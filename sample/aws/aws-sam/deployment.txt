#Step 1 - Create a bucket in the same region as the stack <stackname>
#Make sure that the Region for this bucket aligns with where you deploy
aws s3 mb s3://<bucketname> --region <region>

#Step 2 - Build your application
cd <project-folder>
mvn package shade:shade
sam build

#Step 3 - Package your application
sam package --output-template packaged.yaml --s3-bucket <bucketname> --template-file ./.aws-sam/build/template.yaml

#Step 4 - Deploy your application
sam deploy --template-file packaged.yaml --region <region> --capabilities CAPABILITY_IAM --stack-name <stackname> --parameter-overrides LogRetentionPeriod=7
